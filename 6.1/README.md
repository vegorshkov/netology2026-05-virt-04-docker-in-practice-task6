В итоге (Задача 6.1):
1.Скачан образ hashicorp/terraform:latest
2.Создан временный контейнер terraform-temp
3.Через docker cp скопирован /bin/terraform на локальную машину
4.Проверена работоспособность бинарника (terraform version)
5.Временный контейнер удалён

# ---------------------------------------------
# 1 Cкачиваем образ Terraform (если ещё не скачан)
# ---------------------------------------------
docker pull hashicorp/terraform:latest

# ---------------------------------------------
# 2 Cоздаём временный контейнер
# ---------------------------------------------
docker create --name terraform-temp hashicorp/terraform:latest

        # Проверяем, что контейнер создан
        docker ps -a | grep terraform-temp

# ---------------------------------------------
# 3 Копируем бинарник /bin/terraform на локальную машину
# ---------------------------------------------
docker cp terraform-temp:/bin/terraform ./terraform

# Проверяем, что файл на месте
ls -lh terraform
file terraform

# ---------------------------------------------
# 4 Делаем бинарник исполняемым и проверяем версию
# ---------------------------------------------
chmod +x terraform
./terraform version

# ---------------------------------------------
# 5 Удаляем временный контейнер
# ---------------------------------------------
docker rm terraform-temp
